git 有四大区域 工作区, 暂存区，本地仓库，远程仓库
git config --global user.name  账号   绑定账号
git config --global user.email  邮箱  绑定邮箱
ls 显示当前文件夹的所有文件
mkdir 文件名 创建文件夹
touch 文件名  创建文件
vi 文件名.后缀名 创建文件或插入
git init 创建仓库
cat 文件名 查看文件
head 文件名 查看文件
git add 提交到暂存区
git commit -m "注释" 文件名 将文件提交到本地仓库
git status 查看是否有文件未提交


git add * 把所有的文件提交到暂存区
git commit -m  "注释"   *  将所有文件提交到本地仓库
git add . 把当前目录的文件提交到暂存区
git commit  -m“注释 .   把当前目录的文件提交到本地仓库
git diff 文件名 文件是否被修改  返回红色为删减，绿色为添加
git diff * 把所有被修改的文件显示出来
git rm -f 文件名 删除文件删除后进行注释（git commit  -m "注释" ）
git log 查看所有提交记录
git reset haed^ 回退到上一个版本
git reflog 查看所有版本号
git reset --hard  版本号 回退到指定版本
git checkout 文件名 从仓库取出最新的文件覆盖本地的文件

git remote add 远程仓库名称 路径 连接远程数据库
git push --force 远程仓库名称 master  文件上传远程数据库
git remote 查看远程仓库
git pull 远程仓库名称 master
git branch 名称 创建分支
git branch 查看所有分支
git checkout 分支名 切换分支
git merge 分支名 合并分支
git branch -d 分支名 删除分支
git tag -a  自定义名字 创建标签库
